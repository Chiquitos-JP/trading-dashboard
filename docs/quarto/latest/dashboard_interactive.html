from pathlib import Path
import pandas as pd
import re

# Try candidate relative paths from this working directory (quarto file folder)
cwd = Path.cwd()
candidate_paths = [cwd.joinpath('..','..','..','04_output','figures'), cwd.joinpath('..','..','04_output','figures'), cwd.joinpath('..','04_output','figures')]
base_dir = None
for p in candidate_paths:
    p = p.resolve()
    if p.exists() and any(p.iterdir()):
        base_dir = p
        break

if base_dir is None:
    raise FileNotFoundError(f"Could not find 04_output/figures from {cwd}; tried: {[str(x.resolve()) for x in candidate_paths]}")

# find realizedPl_* directories and pick the latest (alphabetic sort is OK as they are timestamped)
realized_dirs = sorted([d for d in base_dir.iterdir() if d.is_dir() and d.name.startswith('realizedPl_')])
if not realized_dirs:
    raise FileNotFoundError(f"No realizedPl_* directories found in {base_dir}")
latest_dir = realized_dirs[-1]

# find the kpi CSV inside latest_dir (pattern-based)
csv_candidates = list(latest_dir.glob('kpi_ytd_monthly*.csv'))
if not csv_candidates:
    raise FileNotFoundError(f"No kpi_ytd_monthly_*.csv found in {latest_dir}")
csv_path = csv_candidates[-1]

# read
df = pd.read_csv(csv_path)

def _parse_numeric(x):
    if pd.isna(x):
        return None
    if isinstance(x, (int, float)):
        return x
    s = str(x).strip()
    # missing / dash
    if s in ['', '—', '-']:
        return None
    # percent
    if s.endswith('%'):
        try:
            return float(s.strip('%')) / 100.0
        except:
            return s
    # remove commas and currency symbols
    s2 = re.sub(r'[¥$,]', '', s)
    try:
        return float(s2)
    except:
        return s

def get_metric(metric, col='YTD in 2025', parse=True):
    row = df[df['Metric'] == metric]
    if row.empty:
        return None
    val = row.iloc[0].get(col, None)
    return _parse_numeric(val) if parse else val

# extract common metrics
total_net_pl = get_metric('Total Net P/L (JPY)')
# Nov-2025 may or may not be present; attempt to read last month column
monthly_col_candidates = [c for c in df.columns if re.search(r'Nov|Dec|Oct|Sep|2025', c)]
monthly_profit = None
if 'Nov-2025' in df.columns:
    monthly_profit = _parse_numeric(df[df['Metric']=='Total Net P/L (JPY)'].iloc[0].get('Nov-2025'))
else:
    # pick last numeric month for 2025 by header order
    for col in reversed(df.columns):
        if re.search(r'2025', col) and col != 'YTD in 2025':
            monthly_profit = _parse_numeric(df[df['Metric']=='Total Net P/L (JPY)'].iloc[0].get(col))
            break

win_rate = get_metric('Win Rate (WR)', parse=False)
total_trades = get_metric('Total Trades')
roi = get_metric('ROI', parse=False)
rrr = get_metric('Risk/Reward Ratio (RRR)')
pat = get_metric('Profit After Tax (PAT)')

# compute max drawdown as minimum of cumulative net P/L across 2025 columns where numeric
net_row = df[df['Metric']=='Total Net P/L (JPY)']
max_drawdown = None
if not net_row.empty:
    values = []
    for col in df.columns:
        if re.search(r'2025', col) and col != 'YTD in 2025':
            v = _parse_numeric(net_row.iloc[0].get(col))
            if isinstance(v, (int, float)):
                values.append(v)
    if values:
        max_drawdown = min(values)

import re
import plotly.graph_objects as go

# build month columns for 2025 in document order
month_cols = [c for c in df.columns if re.search(r'-2025', c) and c != 'YTD in 2025']
def month_key(col):
    m = re.search(r'([A-Za-z]+)-2025', col)
    if m:
        names = {'Jan':1,'Feb':2,'Mar':3,'Apr':4,'May':5,'Jun':6,'Jul':7,'Aug':8,'Sep':9,'Oct':10,'Nov':11,'Dec':12}
        k = col.split('-')[0][:3]
        return names.get(k, 0)
    m2 = re.search(r'(\d{1,2})', col)
    return int(m2.group(1)) if m2 else 0

month_cols_sorted = sorted(month_cols, key=month_key)

# cumulative trend (Total Net P/L)
net_row = df[df['Metric']=='Total Net P/L (JPY)']
cum_vals = [ _parse_numeric(net_row.iloc[0].get(c)) or 0 for c in month_cols_sorted ]
months = [c.replace('-2025','') for c in month_cols_sorted]

fig = go.Figure()
fig.add_trace(go.Scatter(x=months, y=cum_vals, mode='lines+markers', name='Total Net P/L (JPY)'))
fig.update_layout(title='累積資本利益 (2025)', width=900, height=420)
fig.show()
