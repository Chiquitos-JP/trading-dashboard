Overview

This week’s TidyTuesday explores the major market events of 2025, visualizing how the S&P 500 and Nikkei 225 responded to key events like tariff announcements, Fed rate decisions, and geopolitical tensions.

The visualization features:

  • Dual-axis line chart: S&P 500 and Nikkei 225 normalized for comparison
  • Trend shading: Green for uptrends, red for downtrends
  • Event markers: Vertical lines marking significant market events

Dataset

Source: Custom dataset compiled from Reuters, FRED, and market data

import pandas as pd
import numpy as np
from pathlib import Path
import plotly.graph_objects as go
from plotly.subplots import make_subplots
# Navigate to project root
base_path = Path.cwd()
while not (base_path / "data").exists() and base_path.parent != base_path:
    base_path = base_path.parent

events_path = base_path / "data/macro_economy/market_events/events_2025.csv"
trends_path = base_path / "data/macro_economy/market_events/trend_periods_2025.csv"

events = pd.read_csv(events_path)
events['date'] = pd.to_datetime(events['date'])

trends = pd.read_csv(trends_path)
trends['start_date'] = pd.to_datetime(trends['start_date'])
trends['end_date'] = pd.to_datetime(trends['end_date'])

# Load S&P 500 from FRED parquet

    sp500 = pd.read_parquet(sp500_path)

Visualization

Main Chart: S&P 500 with Events and Trends

# Filter US trends
us_trends = trends[trends['market'] == 'US']

# Filter US/BOTH events
us_events = events[events['market'].isin(['US', 'BOTH']) & events['sp500_change'].notna()]

# Create main visualization
fig = go.Figure()

# Add trend shading
for _, trend in us_trends.iterrows():
    color = 'rgba(34, 197, 94, 0.15)' if trend['trend_type'] == 'UP' else 'rgba(239, 68, 68, 0.15)'
    fig.add_vrect(
        x0=trend['start_date'].strftime('%Y-%m-%d'),
        x1=trend['end_date'].strftime('%Y-%m-%d'),
        fillcolor=color,
        layer='below',
        line_width=0
    )

# Add S&P 500 line
fig.add_trace(go.Scatter(
    x=index_data['date'],
    y=index_data['sp500'],
    mode='lines',
    name='S&P 500',
    line=dict(color='#2563eb', width=2)
))

# Add event markers
for _, event in us_events.iterrows():
    fig.add_vline(
        x=event['date'].strftime('%Y-%m-%d'),
        line_dash='dash',
        line_color='#64748b',
        opacity=0.7
    )

# Add event annotations at top
for i, (_, event) in enumerate(us_events.iterrows()):
    fig.add_annotation(
        x=event['date'].strftime('%Y-%m-%d'),
        y=1.02,
        yref='paper',
        text=event['event_label'],
        showarrow=False,
        textangle=-90,
        font=dict(size=9, color='#334155'),
        xanchor='center',
        yanchor='bottom'
    )

# Update layout
fig.update_layout(
    title=dict(
        text='<b>2025 Market Events: S&P 500</b><br><span style="font-size:12px;color:#64748b;">Trend periods (shaded) and major events (dashed lines)</span>',
        x=0.5,
        xanchor='center'
    ),
    xaxis_title=None,
    yaxis_title='S&P 500 Index',
    template='plotly_white',
    height=500,
    showlegend=True,
    legend=dict(orientation='h', yanchor='bottom', y=1.02, xanchor='right', x=1),
    margin=dict(t=120, r=20, b=50, l=60)
)

fig.update_xaxes(
    tickformat='%b',
    dtick='M1',
    ticklabelmode='period'
)

fig.update_yaxes(
    tickformat=','
)

fig.show()

Event Impact Summary

# Bar chart of event impacts
event_impacts = events[events['sp500_change'].notna()].copy()
event_impacts['direction'] = event_impacts['sp500_change'].apply(lambda x: 'Positive' if x > 0 else 'Negative')
event_impacts = event_impacts.sort_values('sp500_change')

fig2 = go.Figure()

fig2.add_trace(go.Bar(
    x=event_impacts['sp500_change'],
    y=event_impacts['event_label'],
    orientation='h',
    marker_color=event_impacts['direction'].map({'Positive': '#22c55e', 'Negative': '#ef4444'})
))

fig2.update_layout(
    title=dict(
        text='<b>S&P 500 Daily Change by Event</b><br><span style="font-size:12px;color:#64748b;">Impact of major 2025 market events</span>',
        x=0.5,
        xanchor='center'
    ),
    xaxis_title='Daily Change (%)',
    yaxis_title=None,
    template='plotly_white',
    height=400,
    showlegend=False,
    margin=dict(t=80, r=20, b=50, l=200)
)

fig2.show()

Key Findings

  1. Tariff Shock (April 2025): The reciprocal tariffs announcement caused the largest single-day drop (-4.88%), followed by a historic +9.5% rally when the 90-day pause was announced.

  2. Fed Easing Cycle: Three rate cuts (September, October, December) supported a gradual uptrend in the second half of 2025.

  3. Geopolitical Events: The Israel-Iran escalation in June triggered a brief risk-off period with oil spiking +7%.

  4. Trend Analysis: The year showed alternating patterns of sharp corrections followed by sustained rallies.
